    $(document).ready(function() {
      fetch('https://playourshiny.duckdns.org/songdatabase', { mode: 'cors' })
        .then(response => {
          if (!response.ok) {
            throw new Error('API response failed');
          }
          return response.json();
        })
        .then(data => {
          const tableBody = $('#flaskBody');
          const rows = data.map(row => {
            const tableRow = $('<tr><td>' +
              row.id + '</td><td>' +
              row.title + '</td><td>' +
              row.artist + '</td><td>' +
              row.top_genre + '</td><td>' +
              row.year + '</td><td>' +
              row.bpm + '</td><td>' +
              row.energy + '</td><td>' +
              row.danceability + '</td><td>' +
              row.loudness + '</td><td>' +
              row.liveness + '</td><td>' +
              row.valence + '</td><td>' +
              row.length + '</td><td>' +
              row.acousticness + '</td><td>' +
              row.speechiness + '</td><td>' +
              row.popularity + '</td></tr>');
            tableBody.append(tableRow.hide());
            return tableRow;
          });

          let index = 0;
          function animateNextRow() {
            const row = rows[index];
            row.fadeIn(500);
            index++;
            if (index < rows.length) {
              setTimeout(animateNextRow, 100);
            }
          }

          animateNextRow();

          $("#flaskTable").DataTable();
        })
        .catch(error => {
          console.error('Error:', error);
        });
    });
  </script>
